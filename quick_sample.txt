SELECT d.*
FROM data d
JOIN user u ON (
    d.name LIKE CONCAT('%', SUBSTRING_INDEX(u.name, ' ', 1), '%')
    OR d.name LIKE CONCAT('%', SUBSTRING_INDEX(u.name, ' ', -1), '%')
)
WHERE u.dept = 't';
